const raspberryModel = require('./models')

// On cr√©√©e une instance d'express "app"
const app = express()

  res.send('Hello World! ‚ù§Ô∏è‚ù§Ô∏è‚ù§Ô∏è‚ù§Ô∏èüò≠üëåüíªüëåüëåüëå')

// Lors d'un get sur la route /leds
// On cherche √† obtenir depuis notre raspberry l'√©tat des leds de ce dernier
// la fonction est asynchrone, car on attend les actions avec la bdd
app.get('/leds', async(req, res) => {
    // On r√©cup√®re le param√®tre mac pass√© en get
    let mac = req.query.mac
    
    if(mac){
        // On v√©rifie s'il existe d√©j√† en bdd, sinon on le cr√©√© avec des valeurs
        // par d√©faut
        let raspberry = await raspberryModel.findOne({"mac":mac})
        if(!raspberry){
            raspberry = await raspberryModel.create(
                {
                "mac":mac,
                "nom":mac, 
                etatLeds:[,
                    {
                        label:"Jaune",
                        etat:true
                    },
                    {
                        label:"Vert",
                        etat:true
                    },
                    {
                        label:"Rouge",
                        etat:true
                    },
                ]})
        }
        // on met √† jour la date de derni√®re connexion
        raspberry.dateDerniereConnexion=Date.now()
        // On sauvegarde notre objet (√† jour)
        await raspberry.save()
        // on retourne le tableau etatLeds pour traitement par le raspberry
        let result = raspberry.etatLeds
        res.send(result)
    }else{
        res.send({'result':null})
    }    
})

// R√©cup√®re l'ensemble des raspberrys pr√©sents en base
app.get('/raspberries', async(req,res)=>{
    let raspberries = await raspberryModel.find()
    res.send(raspberries)
})

// On r√©cup√®re un raspberry en particulier √† partir de son id pass√© en param√®tre
app.get('/raspberry/:id', async(req,res)=>{
    let id = req.params.id
    let toReturn={}
    
    if(id){
        let raspberry = await raspberryModel.findById(id)
        if(raspberry){
            toReturn=raspberry
        }
    }

    res.send(toReturn)
})

// On met √† jour l'√©tat des LEDs d'un raspberry
// L'id du raspberry √† modifier est pass√© en param
// le contenu d'etatLeds (contient les 3 valeurs des leds) est pass√© dans le body du post
// au format json
app.post('/raspberry/:id/ledsChange', async (req, res)=>{
    // Je recherche le raspberry en bdd
    let raspberry = await raspberryModel.findById(req.params.id)
    if(raspberry){
        // Je remplace la propri√©t√© etatLeds de l'objet raspberry par le JSON envoy√© en post
        raspberry.etatLeds=req.body
        // mongoose ne d√©tecte pas nativement les modification "deep" il faut lui indiquer ce que l'on a modifi√©
        raspberry.markModified('etatLeds')
        // on sauvegarde en base nos modifs
        await raspberry.save()
    }
})
